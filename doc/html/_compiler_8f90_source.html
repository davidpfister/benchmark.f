136 subroutine get_compilerinfo(os)
137 integer, intent(in), optional :: os
138 !private
139 integer :: compiler, os_, i, j, k, l
140 character(:), allocatable :: version
141 character(:), allocatable :: options
142 type(compilerinfo) :: ci
143
144 if (.not. present(os)) then
145 os_ = get_os_type()
146 else
147 os_ = os
148 end if
149
150 version = to_lower(compiler_version())
151
152 if (index(version, 'gcc') > 0) then
153 compiler = compiler_gcc
154 else if (index(version, 'intel') > 0) then
155 if (index(version, 'classic') > 0) then
156 if (os_ == os_linux) then
157 compiler = compiler_intel_classic_nix
158 elseif (os_ == os_windows) then
159 compiler = compiler_intel_classic_windows
160 elseif (os_ == os_macos) then
161 compiler = compiler_intel_classic_mac
162 end if
163 else
164 if (os_ == os_linux) then
165 compiler = compiler_intel_llvm_nix
166 elseif (os_ == os_windows) then
167 compiler = compiler_intel_llvm_windows
168 else
169 compiler = compiler_intel_llvm_unknown
170 end if
171 end if
172 else if (index(version, 'nvfortran') > 0) then
173 compiler = compiler_nvhpc
174 else if (index(version, 'nag fortran') > 0) then
175 compiler = compiler_nag
176 else if (index(version, 'flang') > 0) then
177 compiler = compiler_flang
178 else if (index(version, 'ibm xl') > 0) then
179 compiler = compiler_ibmxl
180 else if (index(version, 'lahey/fujitsu') > 0) then
181 compiler = compiler_lahey
182 else if (index(version, 'lfortran') > 0) then
183 compiler = compiler_lfortran
184 else
185 compiler = get_compiler_at_compiletime(os_)
186 end if
187
188 ci = build_compilerinfo(compiler, version)
189 options = compiler_options()
190 i = 1
191 k = 1
192 write(output_unit,'(A)') ' Name: '//trim(ci%name)
193 write(output_unit,'(A)') ' Vendor: '//trim(ci%vendor)
194 write(output_unit,'(A)') ' Version: '//trim(ci%version)
195 do while (i < len(options))
196 k = index(options(i:), ' ')
197 if (k > 0) then
198 do while (k < 60)
199 if (j == i + k) exit
200 j = i + k
201 l = index(options(i + k + 1:), ' ')
202 if (l > 0) then
203 k = k + l
204 else
205 k = k + len(options(i + k + 1:))
206 j = i + k
207 exit
208 end if
209 end do
210 if (i == j + 1) j = i + 60
211 else
212 j = len(options)
213 end if
214
215 if (i == 1) then
216 write(output_unit,'(A)') ' Options: '//options(i:j)
217 else
218 write(output_unit,'(A)') ' '//options(i:j)
219 end if
220 i = j + 1
221 k = 0
222 end do
223
224 associate(isdebug => compiler_is_debug(compiler, options))
225 write(output_unit,'(A)') ' Profile: '//trim(merge('Debug ', 'Release', isdebug))
226 if (isdebug) then
227 call warning_debug()
228 end if
229 end associate

234 character(*), intent(in) :: input
235 character(len(input)) :: str
236 !private
237 integer, parameter :: change_case = 32
238 integer :: i
239
240 do i = 1, len(input)

244 case default

246 end select

292 logical function compiler_is_debug(compiler, options) result(is)
293 integer, intent(in) :: compiler
294 character(*), intent(in) :: options
295
296 is = .false.
297 if (index(options, '-O0') > 0) then
298 is = .true.
299 return
300 end if
301 select case (compiler)
302 case (compiler_ibmxl)
303 if (index(options, '-qnoopt') > 0) then
304 is = .true.
305 return
306 end if
307 case (compiler_intel_classic_nix, compiler_intel_classic_mac, compiler_intel_llvm_nix, compiler_intel_llvm_unknown)
308 if (index(options, '-debug') > 0) then
309 is = .true.
310 return
311 end if
312 case (compiler_intel_classic_windows, compiler_intel_llvm_windows)
313 if (index(options, '/debug') > 0 .or. index(options, '/Od') > 0) then
314 is = .true.
315 return
316 end if
317 case (compiler_lfortran)
318 if (index(options, '--debug-with-line-column')> 0) then
319 is = .true.
320 return
321 end if
322 end select
323 end function

325 type(compilerinfo) function build_compilerinfo(id, string) result(info)
326 integer, intent(in) :: id
327 character(*), intent(in) :: string
328 !private
329 integer :: i, j
330

332 case (compiler_unknown)

334 case (compiler_gcc)
335 i = index(string, 'version')
336 info = compilerinfo(id, 'gfortran', 'GCC', trim(adjustl(string(i+7:))))
337 case (compiler_f95)
338 info = compilerinfo(id, 'f95', 'ORACLE', 'X.X.X.X')
339 case (compiler_caf)
340 info = compilerinfo(id, 'caf', 'OPENCOARRAY', 'X.X.X.X')
341 case (compiler_intel_classic_nix)
342 i = index(string, 'version')
343 j = index(string, 'build')
344 info = compilerinfo(id, 'ifort', 'INTEL', trim(adjustl(string(i+7:j-1))))
345 case (compiler_intel_classic_mac)
346 i = index(string, 'version')
347 j = index(string, 'build')
348 info = compilerinfo(id, 'ifort','INTEL', trim(adjustl(string(i+7:j-1))))
349 case (compiler_intel_classic_windows)
350 i = index(string, 'version')
351 j = index(string, 'build')
352 info = compilerinfo(id, merge('ia32 ', 'ifort', index(string, 'IA-32') > 0),'INTEL', trim(adjustl(string(i+7:j-1))))
353 case (compiler_intel_llvm_nix)
354 i = index(string, 'version')
355 j = index(string, 'build')
356 info = compilerinfo(id, 'ifx', 'INTEL', trim(adjustl(string(i+7:j-1))))
357 case (compiler_intel_llvm_windows)
358 i = index(string, 'version')
359 j = index(string, 'build')
360 info = compilerinfo(id, 'ifx', 'INTEL', trim(adjustl(string(i+7:j-1))))
361 case (compiler_intel_llvm_unknown)
362 i = index(string, 'version')
363 j = index(string, 'build')
364 info = compilerinfo(id, 'ifx', 'INTEL', trim(adjustl(string(i+7:j-1))))
365 case (compiler_pgi)
366 info = compilerinfo(id, 'pgfortran','PGI', 'X.X.X.X')
367 case (compiler_nvhpc)
368 i = index(string, 'nvfortran')
369 j = index(string, 'llvm')
370 info = compilerinfo(id, 'nvhpc','NVIDIA', trim(adjustl(string(i+9:j-1))))
371 case (compiler_nag)
372 i = index(string, 'release')
373 j = index(string, '(')
374 info = compilerinfo(id, 'nagfor', 'NAG', trim(adjustl(string(i+7:j-1))))
