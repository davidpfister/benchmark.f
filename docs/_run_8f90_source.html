53 subroutine step_run(step)
54 class(workflow), intent(inout) :: step
55 !private
56 integer :: k
57 real(r8) :: start, finish
58 type(stats) :: s
59 real(r8), allocatable :: times(:)
60 real(r8) :: treshold
61
62
63 maxcall_reached = .false.
64 select type(step)
65 type is (benchmark_run)
66 treshold = step%options%ssd_threshold
67 if (first_call) write (output_unit, '(A)') new_line('A'), step%header, new_line('A')
68
69 step%options%count = step%options%count + 1
70 allocate(times(step%options%sampling_window), source=0.0_r8)
71
72 block
73 integer :: icount, jcount, count
74 real(r8) :: crit, t
75
76 crit = 0.0_r8; icount = 0; jcount = 0; count = 0
77 call clock(start)
78 do while (crit <= (1.0_r8 - treshold))
79 call step%method%invoke(); call clock(finish)
80 t = finish - start
81 icount = icount + 1
82 if (t > step%options%mintime) then
83 times(1 + modulo(jcount, step%options%sampling_window)) = t / real(icount, r8)
84 jcount = jcount + 1
85
86 if (jcount >= step%options%sampling_window) crit = ssd(times, modulo(jcount, step%options%sampling_window), treshold)
87
88 icount = 0
89 call clock(start)
90 end if
91 count = count + 1
92
93 if (count >= step%options%maxcalls) then
94 maxcall_reached = .true.
95 display_maxcall_warning = .true.
96 exit
97 end if
98 end do
99 end block
100
101 call s%compute(times)
102 if (step%options%csv_unit /= 0) then
103 call summary(step, s, step%options%csv_unit)
104 else
105 call summary(step, s)
106 end if
107 deallocate(times)
108 nullify(step%method)
109 nullify(step%options)
110 end select
111 end subroutine
