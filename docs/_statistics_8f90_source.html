28 !! <h2>Remarks</h2>

62 !> Compute mean, standard deviation and
63 !! variance of a Type bound to the method
64 !! @param[inout] this bound argument
65 !! @param[in] y Sample array
66 !!
67 !! @b Examples
68 !!
69 !! @code{.f90}
70 !! type(stats) :: s
71 !! integer :: i
72 !! real(r8) :: array(10)
73 !!
74 !! call random_number(array)
75 !! call s%compute(array)
76 !! @endcode
77 !! @note At the moment it only works for real(r8) arrays
78 !!
79 !! @ingroup group_statistics
80 !! @b Remarks
81 subroutine stats_compute(this, y)
82 class(stats), intent(inout) :: this
83 real(r8), intent(in) :: y(:)
84
85 this%n = size(y)
86 this%mean = sum(y) / real(this%n, r8)
87 this%min = minval(y)
88 this%max = maxval(y)
89 this%variance = sum((y - this%mean)**2) / real(this%n - 1, r8)
90 this%stddev = sqrt(this%variance)
91 end subroutine
