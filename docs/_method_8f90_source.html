838 subroutine method_assign_method(lhs, rhs)
839 class(method), intent(inout) :: lhs
840 type(method), intent(in) :: rhs
841 !private
842 integer :: i
843
844 nullify(lhs%f)
845 nullify(lhs%caller)
846 lhs%f => rhs%f
847 if (associated(rhs%caller)) then
848 lhs%caller => rhs%caller
849 end if
850 lhs%nargs = rhs%nargs
851 if (allocated(lhs%args)) deallocate(lhs%args)
852 allocate(lhs%args(rhs%nargs), source = rhs%args)
853 !
