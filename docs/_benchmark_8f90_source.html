599 function parse_names(this, n, name) result(names)
600 class(runner), intent(inout) :: this
601 integer, intent(in) :: n
602 character(*), intent(in), optional :: name
603 !private
604 type(string) :: names(n)
605 character(:), allocatable :: func
606 character(:), allocatable :: substr
607 integer :: i, j, k, l
608
609 if (present(name)) then
610 this%name = name
611 else
612 l = len_trim(this%name)
613 if (len_trim(this%name) == 0) return
614 i = index(this%name, ',', back = .true.)
615 if (i > 0) then
616 k = index(this%name, ')', back = .true.)
617 func = this%name(i+1:merge(k-1, l, k/=0))
618 else
619 i = index(this%name, '(', back = .false.)
620 k = index(this%name, ')', back = .true.)
621 this%name = this%name(i+1:merge(k-1, l, k/=0))
622 return
623 end if
624 j = n
625 do while (i > 0)
626 substr = this%name(:i-1)
627 i = index(substr, ',', back = .true.)
628 if (i > 0) then
629 names(j) = substr(i+1:)
630 j = j - 1
631 else
632 exit
633 end if
634 end do
635 i = index(this%name, '(', back = .false.)
636 if (i > 0) then
637 names(1) = this%name(i+1:index(this%name, ',', back = .false.)-1)
638 this%name = func
639 return
640 else
641 i = index(this%name, ' ', back = .false.)
642 names(1) = this%name(i+1:index(this%name, ',', back = .false.)-1)
643 this%name = func
644 return
645 end if
646 end if
647 end function
