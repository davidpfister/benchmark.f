28 !> Compute the probability that a sample `x` reached steady-state.
29 !! The SSD algorithm presented in this work is also window-based and utilizes the
30 !! Student-t test to determine if the difference between the process signal value
31 !! minus its mean is above or below the standard-deviation times its statistical critical value. If less
32 !! than, then that time instant or point is steady and if greater than, then it is unsteady
33 !! where the aggregation is computed over the window approximating a probability or
34 !! frequency of being at steady-state.
35 !! @param[in] x Sample of real(r8) values
36 !! @param[in] offset Index of the first values inside the periodic array x
37 !! @param[in] alpha Significance level for the student test.
38 !!
39 !! The mean is defined as:
40 !! @f[
41 !! \mu = \frac{1}{n}\left(\sum x-m\sum i\right)
42 !! @f]
43 !! where @f$m@f$ is the slope of a linear drift. In the present case, @f$m=0@f$
44 !! It comes that the standard deviation is given by
45 !! @f[
46 !! \sigma = \sqrt{\frac{1}{n-2}\sum (x-\mu)^2}
47 !! @f]
48 !! At this point along with a specified Student-t critical or threshold value at a particu100 lar significance level α and degrees-of-freedom n, all of the necessary information is
49 !! available to test the null-hypothesis that the process signal is steady or is stationary
50 !! about @f$\mu@f$
51 !! @f[
52 !! if\ x - \mu < t_{crit}-\sigma\ , then\ 1,\ else\ 0
53 !! @f]
54 !! @f$t_{crit}@f$ is evaluated for a given significance level and a given degree of freedom
55 !! using the t-distribution.
56 !!
57 !! group_ssd
58 !! @b Remarks
59 real(r8) function ssd(x, offset, alpha) result(res)
60 real(r8), intent(in) :: x(:)
61 integer, intent(in) :: offset
62 real(r8), intent(in) :: alpha
63 !private
64 integer :: j, n
65 real(r8) :: mu, sd
66 real(r8) :: y(size(x))
67
68 n = size(y)
69 y(1:n - offset) = x(offset + 1:n)
70 y(n - offset + 1:) = x(:offset)
71
72 mu = sum(y) / real(n, r8)
73
74 sd = 0.0_r8
75 do j = 1, n
76 sd = sd + (y(j) - mu)**2
77 end do
78 sd = sqrt(sd / real(n - 2, r8))
79
80 res = 0.0_r8
81 associate(t_crit => student(n, alpha))
82 do j = 1, n
83 if (abs(y(j) - mu) <= t_crit * sd) res = res + 1.0_r8
84 end do
85 end associate
86 res = res / real(n, r8)
87
88 end function

91 real(r8) function student(dof, a) result(t)
92 integer, intent(in) :: dof
93 real(r8), intent(in) :: a
94 !private
95 real(r8) :: f, fold, x, p, integral
96 real(r8), parameter :: dt = 1.0e-2_r8
97
98 f(x) = (1 + x**2 / real(dof, r8))**(-(dof + 1) / 2.0_r8) * gamma((dof + 1) / 2.0_r8) / sqrt(3.14_r8 * dof) / gamma(dof / 2.0_r8)
99
100 integral = 0.0_r8
101 fold = 0.0_r8
102 p = 0.5_r8 * a
103 t = 1.0e3_r8
104
105 do while (f(t) > 1.0e-20_r8)
106 t = t * 2
107 end do
108
109 do while (integral < p)
110 associate(fnew => f(t + dt))
111 integral = integral + 0.5_r8 * (fold + fnew) * dt
112 t = t - dt
113 fold = fnew
114 if (t < 0) stop
115 end associate
116 end do
