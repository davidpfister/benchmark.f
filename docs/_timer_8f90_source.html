71 !> @private
72 subroutine get_time(ctime, option) ! in milliseconds
73 real(r8), intent(out) :: ctime
74 integer(CLOCK_ENUM), intent(in) :: option
75
76 select case (option)
77 case (cputime)
78 call cpu_time(ctime)
79 ctime = ctime * 1000_r8
80 case (datetime)
81 block
82 integer(i8) :: dt(8)
83
84 call date_and_time(values=dt)
85 ctime = (dt(5) * 3600_r8 + dt(6) * 60_r8 + dt(7)) * 1000_r8 + dt(8) * 1_r8
86 end block
87 case (systemtime)
88 block
89 integer(i8) :: clock_max, clock_rate, clock_reading
90
91 call system_clock(clock_reading, clock_rate, clock_max)
92 ctime = 1000_r8 * real(clock_reading, r8) / real(clock_rate, r8)
93 end block
94 end select
95 end subroutine
