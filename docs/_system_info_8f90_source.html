58 !> Generate and display a summary of the operating
59 !! system and platform specifications. The system information is obtained by first
60 !! identifying the operating system and the runtime shell and then calling one of
61 !! the following commands:
62 !! - Darwin (macOS): The commands `uname -a` + `sysctl -a | grep machdep.cpu` + `system_profiler SPHardwareDataType`.
63 !! - Linux: The commands `uname -a + lscpu` + `cat /proc/cpuinfo`.
64 !! - Windows: The commands `systeminfo`.
65 !!
66 !! @ingroup group_systeminfo
67 !! @b Remarks
68 subroutine get_systeminfo()
69 !private
70 character(:), allocatable :: tmpout, output, cmsg
71 character(:), allocatable :: cmd
72 integer(OS_ENUM) :: os
73 integer :: lu, ios
74 integer :: ierr, cstat
75 character(200) :: line
76 logical :: exists
77
78 allocate(character(256) :: cmsg)
79
80 tmpout = 'sys.info'
81 inquire(file=tmpout, exist=exists)
82
83 if (.not. exists) then
84 open(newunit=lu, status='new', file=tmpout); close(lu)
85 output = " 1> " // tmpout
86 ierr = 0
87
88 os = get_os_type()
89 ! Define the shell command.
90 if (os == os_macos) then
91 cmd = 'uname -a ' // output // '; sysctl -a | grep machdep.cpu ' // output // '; system_profiler SPHardwareDataType ' // output
92 elseif (os == os_linux) then
93 cmd = 'uname -a ' // output // '; lscpu | grep -v "Not affected" | grep -v "Flags"' // output
94 elseif (os == os_windows) then
95 cmd = 'systeminfo | find /V /I "hotfix" | find /V /I "network"' //&
96 '| find /V "Connection Name" | find /V "[" | find /V "DHCP" ' //&
97 '| find /V "Status" | find /V "IP address" | find /V "Hyper-V" ' // output
98 else
99 return
100 end if
101
102 call execute_command_line(cmd, wait=.true., exitstat=ierr, cmdstat=cstat, cmdmsg=cmsg)
103 if (ierr /= 0) then
104 write(output_unit, *) cmd
105 write(output_unit, *) 'exitstat: ', ierr
106 write(output_unit, *) 'cmdstat: ', cstat
107 write(output_unit, *) 'cmdmsg: ', cmsg
108 return
109 end if
110 end if
111
112 open(newunit=lu, status='old', file=tmpout)
113 do
114 read(lu, '(A)', iostat=ios) line
115 if (ios /= 0) exit
116 if (len_trim(line) > 0) write(output_unit, '(A)') ' ' // trim(line)
117 end do
118
119 !close(lu, status = 'delete')
120 end subroutine

122 !> Returns the OS type.
123 !! At first, the environment variable `OS` is checked, which is usually
124 !! found on Windows. Then, `OSTYPE` is read in and compared with common
125 !! names. If this fails too, check the existence of files that can be
126 !! found on specific system types only.
127 !! @returns An integer(OS_ENUM). Returns OS_UNKNOWN if the operating system cannot be determined.
128 !!
129 !! @ingroup group_systeminfo
130 !! @b Remarks
131 integer(OS_ENUM) function get_os_type() result(r)
132 !private
133 character(255) :: val
134 integer :: length, rc
135 logical :: file_exists
136 integer, allocatable, save :: ret
137 !$omp threadprivate(ret, first_run)
138
139 if (allocated(ret)) then
140 r = ret
141 return
142 end if
143 r = os_unknown
144
145 ! Check environment variable `OSTYPE`.
146 call get_environment_variable('OSTYPE', val, length, rc)
147
148 if (rc == 0 .and. length > 0) then
149 ! Linux
150 if (index(val, 'linux') > 0) then
151 r = os_linux
152 ret = r
153 return
154 end if
155
156 ! macOS
157 if (index(val, 'darwin') > 0) then
158 r = os_macos
159 ret = r
160 return
161 end if
162
163 ! Windows, MSYS, MinGW, Git Bash
164 if (index(val, 'win') > 0 .or. index(val, 'msys') > 0) then
165 r = os_windows
166 ret = r
167 return
168 end if
169
170 ! Cygwin
171 if (index(val, 'cygwin') > 0) then
172 r = os_cygwin
173 ret = r
174 return
175 end if
176
177 ! Solaris, OpenIndiana, ...
178 if (index(val, 'SunOS') > 0 .or. index(val, 'solaris') > 0) then
179 r = os_solaris
180 ret = r
181 return
182 end if
183
184 ! FreeBSD
185 if (index(val, 'FreeBSD') > 0 .or. index(val, 'freebsd') > 0) then
186 r = os_freebsd
187 ret = r
188 return
189 end if
190
191 ! OpenBSD
192 if (index(val, 'OpenBSD') > 0 .or. index(val, 'openbsd') > 0) then
193 r = os_openbsd
194 ret = r
195 return
196 end if
197 end if
198
199 ! Check environment variable `OS`.
200 call get_environment_variable('OS', val, length, rc)
201
202 if (rc == 0 .and. length > 0 .and. index(val, 'Windows_NT') > 0) then
203 r = os_windows
204 ret = r
205 return
206 end if
207
208 ! Linux
209 inquire(file='/etc/os-release', exist=file_exists)
210
211 if (file_exists) then
212 r = os_linux
213 ret = r
214 return
215 end if
216
217 ! macOS
218 inquire(file='/usr/bin/sw_vers', exist=file_exists)
219
220 if (file_exists) then
221 r = os_macos
222 ret = r
223 return
224 end if
225
226 ! FreeBSD
227 inquire(file='/bin/freebsd-version', exist=file_exists)
228
229 if (file_exists) then
230 r = os_freebsd
231 ret = r
232 return
233 end if
234 end function

236 !> Return string describing the OS type flag.
237 !! That function was taken from
238 !! <a href="https://github.com/fortran-lang/fpm/blob/main/src/fpm_environment.f90">fpm</a>
239 !! @returns The name of the OS.
240 !!
241 !! @ingroup group_systeminfo
242 !! @b Remarks
243 pure function os_name(os)
244 integer(OS_ENUM), intent(in) :: os
245 !private
246 character(:), allocatable :: os_name
247
248 select case (os)
249 case (os_linux); os_name = "Linux"
250 case (os_macos); os_name = "macOS"
251 case (os_windows); os_name = "Windows"
252 case (os_cygwin); os_name = "Cygwin"
253 case (os_solaris); os_name = "Solaris"
254 case (os_freebsd); os_name = "FreeBSD"
255 case (os_openbsd); os_name = "OpenBSD"
256 case (os_unknown); os_name = "Unknown"
257 case default ; os_name = "Unknown"
258 end select
259 end function
