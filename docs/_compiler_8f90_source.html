1!> @file
2!! @defgroup group_steps_compiler Compiler
3!! Retrieves compiler information at run time as the compiler's name
4!! and version, as well as compilation options. It uses the intrinsic
5!! functions `compiler_options` and `compiler_version` from the
6!! module `iso_fortran_env`
7!! @ingroup group_steps
8!!
9!! Features
10!! - Automatic detection of compiler name, vendor and version at runtime
11!! - Detection of compilation options and debug/release profile
12!! - Support for a wide range of Fortran compilers (gfortran, ifort/ifx, nvfortran, nagfor, flang, etc.)
13!! - Compile-time fallback detection using preprocessor macros
14!! - Consistent output formatting for benchmark reports
15!!
16!! <h2 class="groupheader">Examples</h2>
17!! @code{.f90}
18!! program test_compiler_info
19!! use benchmark_steps_compiler
20!! use benchmark_workflow
21!!
22!! type(workflow) :: w
23!!
24!! call w%add(compiler())
25!! call w%run()
26!! end program
27!! ...
28!! @endcode
29!!
30!! Another example showing direct usage:
31!! @code{.f90}
32!! use benchmark_steps_compiler
33!!
34!! type(compiler) :: ci
35!! character(:), allocatable :: opts
36!!
37!! ci = build_compilerinfo(COMPILER_GCC, compiler_version())
38!! opts = compiler_options()
39!!
40!! print *, 'Compiler:', trim(ci%name), trim(ci%version)
41!! print *, 'Options:', opts
42!! @endcode

162 !> @brief Detects current compiler and prints name, vendor, version, options and profile
163 !! @ingroup group_steps_compiler
164 !! @b Remarks
165 subroutine get_compilerinfo(os)
166 integer, intent(in), optional :: os
167 !private
168 integer :: c, os_, i, j, k, l
169 character(:), allocatable :: version, options
170 type(compiler) :: ci
171
172 if (.not. present(os)) then
173 os_ = get_os_type()
174 else
175 os_ = os
176 end if
177
178 version = to_lower(compiler_version())
179
180 if (index(version, 'gcc') > 0) then
181 c = compiler_gcc
182 else if (index(version, 'intel') > 0) then
183 if (index(version, 'classic') > 0) then
184 if (os_ == os_linux) then
185 c = compiler_intel_classic_nix
186 elseif (os_ == os_windows) then
187 c = compiler_intel_classic_windows
188 elseif (os_ == os_macos) then
189 c = compiler_intel_classic_mac
190 end if
191 else
192 if (os_ == os_linux) then
193 c = compiler_intel_llvm_nix
194 elseif (os_ == os_windows) then
195 c = compiler_intel_llvm_windows
196 else
197 c = compiler_intel_llvm_unknown
198 end if
199 end if
200 else if (index(version, 'nvfortran') > 0) then
201 c = compiler_nvhpc
202 else if (index(version, 'nag fortran') > 0) then
203 c = compiler_nag
204 else if (index(version, 'flang') > 0) then
205 c = compiler_flang
206 else if (index(version, 'ibm xl') > 0) then
207 c = compiler_ibmxl
208 else if (index(version, 'lahey/fujitsu') > 0) then
209 c = compiler_lahey
210 else if (index(version, 'lfortran') > 0) then
211 c = compiler_lfortran
212 else
213 c = get_compiler_at_compiletime(os_)
214 end if
215
216 ci = build_compilerinfo(c, version)
217 options = compiler_options()
218 i = 1
219 k = 1
220 write(output_unit, '(A)') ' Name: ' // trim(ci%name)
221 write(output_unit, '(A)') ' Vendor: ' // trim(ci%vendor)
222 write(output_unit, '(A)') ' Version: ' // trim(ci%version)
223 do while (i < len(options))
224 k = index(options(i:), ' ')
225 if (k > 0) then
226 do while (k < 60)
227 if (j == i + k) exit
228 j = i + k
229 l = index(options(i + k + 1:), ' ')
230 if (l > 0) then
231 k = k + l
232 else
233 k = k + len(options(i + k + 1:))
234 j = i + k
235 exit
236 end if
237 end do
238 if (i == j + 1) j = i + 60
239 else
240 j = len(options)
241 end if
242
243 if (i == 1) then
244 write(output_unit, '(A)') ' Options: ' // options(i:j)
245 else
246 write(output_unit, '(A)') ' ' // options(i:j)
247 end if
248 i = j + 1
249 k = 0
250 end do
251
252 associate(isdebug => compiler_is_debug(c, options))
253 write(output_unit, '(A)') ' Profile: ' // trim(merge('Debug ', 'Release', isdebug))
254 if (isdebug) then
255 call warning_debug()
256 end if
257 end associate

261 pure function to_lower(input) result(str)
262 character(*), intent(in) :: input !< the input to be lowercased
263 character(len(input)) :: str !< the lower case input
264 !private
265 integer, parameter :: change_case = 32
266 integer :: i
267
268 do i = 1, len(input)
269 select case (input(i:i))
270 case ('A':'Z')
271 str(i:i) = char(ichar(input(i:i)) + change_case)
272 case default
273 str(i:i) = input(i:i)
274 end select
275 end do
276 end function
277 end subroutine

282 function get_compiler_at_compiletime(os) result(id)
283 integer, intent(in) :: os

286 id = compiler_unknown
287#ifdef __GFORTRAN__
288 id = compiler_gcc
289#elif defined(__INTEL_COMPILER)
290#if defined(__INTEL_LLVM_COMPILER)

324 !> Determines if the current compilation is in debug mode
325 !! @ingroup group_steps_compiler
326 !! @b Remarks
327 function compiler_is_debug(compiler, options) result(is)
328 integer, intent(in) :: compiler
329 character(*), intent(in) :: options
330 logical :: is
331
332 is = .false.
333 if (index(options, '-O0') > 0) then
334 is = .true.
335 return
336 end if
337 select case (compiler)
338 case (compiler_ibmxl)
339 if (index(options, '-qnoopt') > 0) then
340 is = .true.
341 return
342 end if
343 case (compiler_intel_classic_nix, compiler_intel_classic_mac, compiler_intel_llvm_nix, compiler_intel_llvm_unknown)
344 if (index(options, '-debug') > 0) then
345 is = .true.
346 return
347 end if
348 case (compiler_intel_classic_windows, compiler_intel_llvm_windows)
349 if (index(options, '/debug') > 0 .or. index(options, '/Od') > 0) then
350 is = .true.
351 return
352 end if
353 case (compiler_lfortran)
354 if (index(options, '--debug-with-line-column') > 0) then
355 is = .true.
356 return
357 end if
358 end select
359 end function

361 !> @brief Constructs a compiler type from compiler id and version string

364 function build_compilerinfo(id, string) result(info)
365 integer, intent(in) :: id
366 character(*), intent(in) :: string
367 type(compiler) :: info
368 !private
369 integer :: i, j
370
371 select case (id)
372 case (compiler_unknown)
373 info = compiler(id, 'UNKNOWN', 'UNKNOWN', 'X.X.X.X')
374 case (compiler_gcc)
375 i = index(string, 'version')
376 info = compiler(id, 'gfortran', 'GCC', trim(adjustl(string(i + 7:))))
377 case (compiler_f95)
378 info = compiler(id, 'f95', 'ORACLE', 'X.X.X.X')
379 case (compiler_caf)
380 info = compiler(id, 'caf', 'OPENCOARRAY', 'X.X.X.X')
381 case (compiler_intel_classic_nix)
382 i = index(string, 'version')
383 j = index(string, 'build')
384 info = compiler(id, 'ifort', 'INTEL', trim(adjustl(string(i + 7:j - 1))))
385 case (compiler_intel_classic_mac)
386 i = index(string, 'version')
387 j = index(string, 'build')
388 info = compiler(id, 'ifort', 'INTEL', trim(adjustl(string(i + 7:j - 1))))
389 case (compiler_intel_classic_windows)
390 i = index(string, 'version')
391 j = index(string, 'build')
392 info = compiler(id, merge('ia32 ', 'ifort', index(string, 'IA-32') > 0), 'INTEL', trim(adjustl(string(i + 7:j - 1))))
393 case (compiler_intel_llvm_nix)
394 i = index(string, 'version')
395 j = index(string, 'build')
396 info = compiler(id, 'ifx', 'INTEL', trim(adjustl(string(i + 7:j - 1))))
397 case (compiler_intel_llvm_windows)
398 i = index(string, 'version')
399 j = index(string, 'build')
400 info = compiler(id, 'ifx', 'INTEL', trim(adjustl(string(i + 7:j - 1))))
401 case (compiler_intel_llvm_unknown)

404 info = compiler(id, 'ifx', 'INTEL', trim(adjustl(string(i + 7:j - 1))))
405 case (compiler_pgi)

407 case (compiler_nvhpc)

409 j = index(string, 'llvm')
