135 subroutine get_compilerinfo(os)
136 integer, intent(in), optional :: os
137 !private
138 integer :: compiler, os_, i, j, k, l
139 character(:), allocatable :: version
140 character(:), allocatable :: options
141 type(compilerinfo) :: ci
142
143 if (.not. present(os)) then
144 os_ = get_os_type()
145 else
146 os_ = os
147 end if
148
149 version = to_lower(compiler_version())
150
151 if (index(version, 'gcc') > 0) then
152 compiler = compiler_gcc
153 else if (index(version, 'intel') > 0) then
154 if (index(version, 'classic') > 0) then
155 if (os_ == os_linux) then
156 compiler = compiler_intel_classic_nix
157 elseif (os_ == os_windows) then
158 compiler = compiler_intel_classic_windows
159 elseif (os_ == os_macos) then
160 compiler = compiler_intel_classic_mac
161 end if
162 else
163 if (os_ == os_linux) then
164 compiler = compiler_intel_llvm_nix
165 elseif (os_ == os_windows) then
166 compiler = compiler_intel_llvm_windows
167 else
168 compiler = compiler_intel_llvm_unknown
169 end if
170 end if
171 else if (index(version, 'nvfortran') > 0) then
172 compiler = compiler_nvhpc
173 else if (index(version, 'nag fortran') > 0) then
174 compiler = compiler_nag
175 else if (index(version, 'flang') > 0) then
176 compiler = compiler_flang
177 else if (index(version, 'ibm xl') > 0) then
178 compiler = compiler_ibmxl
179 else if (index(version, 'lahey/fujitsu') > 0) then
180 compiler = compiler_lahey
181 else if (index(version, 'lfortran') > 0) then
182 compiler = compiler_lfortran
183 else
184 compiler = get_compiler_at_compiletime(os_)
185 end if
186
187 ci = build_compilerinfo(compiler, version)
188 options = compiler_options()
189 i = 1
190 k = 1
191 write(output_unit,'(A)') ' Name: '//trim(ci%name)
192 write(output_unit,'(A)') ' Vendor: '//trim(ci%vendor)
193 write(output_unit,'(A)') ' Version: '//trim(ci%version)
194 do while (i < len(options))
195 k = index(options(i:), ' ')
196 if (k > 0) then
197 do while (k < 60)
198 if (j == i + k) exit
199 j = i + k
200 l = index(options(i + k + 1:), ' ')
201 if (l > 0) then
202 k = k + l
203 else
204 k = k + len(options(i + k + 1:))
205 j = i + k
206 exit
207 end if
208 end do
209 if (i == j + 1) j = i + 60
210 else
211 j = len(options)
212 end if
213
214 if (i == 1) then
215 write(output_unit,'(A)') ' Options: '//options(i:j)
216 else
217 write(output_unit,'(A)') ' '//options(i:j)
218 end if
219 i = j + 1
220 k = 0
221 end do
222
223 associate(isdebug => compiler_is_debug(compiler, options))
224 write(output_unit,'(A)') ' Profile: '//trim(merge('Debug ', 'Release', isdebug))
225 if (isdebug) then
226 call warning_debug()
227 end if
228 end associate

233 character(*), intent(in) :: input
234 character(len(input)) :: str
235 !private
236 integer, parameter :: change_case = 32
237 integer :: i
238
239 do i = 1, len(input)

242 str(i:i) = char(ichar(input(i:i)) + change_case)
243 case default
244 str(i:i) = input(i:i)
245 end select
246 end do

291 logical function compiler_is_debug(compiler, options) result(is)
292 integer, intent(in) :: compiler
293 character(*), intent(in) :: options
294
295 is = .false.
296 if (index(options, '-O0') > 0) then
297 is = .true.
298 return
299 end if
300 select case (compiler)
301 case (compiler_ibmxl)
302 if (index(options, '-qnoopt') > 0) then
303 is = .true.
304 return
305 end if
306 case (compiler_intel_classic_nix, compiler_intel_classic_mac, compiler_intel_llvm_nix, compiler_intel_llvm_unknown)
307 if (index(options, '-debug') > 0) then
308 is = .true.
309 return
310 end if
311 case (compiler_intel_classic_windows, compiler_intel_llvm_windows)
312 if (index(options, '/debug') > 0 .or. index(options, '/Od') > 0) then
313 is = .true.
314 return
315 end if
316 case (compiler_lfortran)
317 if (index(options, '--debug-with-line-column')> 0) then
318 is = .true.
319 return
320 end if
321 end select
322 end function

324 type(compilerinfo) function build_compilerinfo(id, string) result(info)
325 integer, intent(in) :: id
326 character(*), intent(in) :: string
327 !private
328 integer :: i, j
329
330 select case (id)
331 case (compiler_unknown)
332 info = compilerinfo(id, 'UNKNOWN', 'UNKNOWN', 'X.X.X.X')
333 case (compiler_gcc)
334 i = index(string, 'version')
335 info = compilerinfo(id, 'gfortran', 'GCC', trim(adjustl(string(i+7:))))
336 case (compiler_f95)
337 info = compilerinfo(id, 'f95', 'ORACLE', 'X.X.X.X')
338 case (compiler_caf)
339 info = compilerinfo(id, 'caf', 'OPENCOARRAY', 'X.X.X.X')
340 case (compiler_intel_classic_nix)
341 i = index(string, 'version')
342 j = index(string, 'build')
343 info = compilerinfo(id, 'ifort', 'INTEL', trim(adjustl(string(i+7:j-1))))
344 case (compiler_intel_classic_mac)
345 i = index(string, 'version')
346 j = index(string, 'build')
347 info = compilerinfo(id, 'ifort','INTEL', trim(adjustl(string(i+7:j-1))))
348 case (compiler_intel_classic_windows)
349 i = index(string, 'version')
350 j = index(string, 'build')
351 info = compilerinfo(id, merge('ia32 ', 'ifort', index(string, 'IA-32') > 0),'INTEL', trim(adjustl(string(i+7:j-1))))
352 case (compiler_intel_llvm_nix)
353 i = index(string, 'version')
354 j = index(string, 'build')
355 info = compilerinfo(id, 'ifx', 'INTEL', trim(adjustl(string(i+7:j-1))))
356 case (compiler_intel_llvm_windows)
357 i = index(string, 'version')
358 j = index(string, 'build')
359 info = compilerinfo(id, 'ifx', 'INTEL', trim(adjustl(string(i+7:j-1))))
360 case (compiler_intel_llvm_unknown)
361 i = index(string, 'version')
362 j = index(string, 'build')
363 info = compilerinfo(id, 'ifx', 'INTEL', trim(adjustl(string(i+7:j-1))))
364 case (compiler_pgi)
365 info = compilerinfo(id, 'pgfortran','PGI', 'X.X.X.X')
366 case (compiler_nvhpc)
367 i = index(string, 'nvfortran')
368 j = index(string, 'llvm')
369 info = compilerinfo(id, 'nvhpc','NVIDIA', trim(adjustl(string(i+9:j-1))))
370 case (compiler_nag)
371 i = index(string, 'release')
372 j = index(string, '(')
373 info = compilerinfo(id, 'nagfor', 'NAG', trim(adjustl(string(i+7:j-1))))
374 case (compiler_flang)
375 i = index(string, 'version')
376 info = compilerinfo(id, 'flang', 'FLANG', trim(adjustl(string(i+7:))))
377 case (compiler_flang_new)
378 i = index(string, 'version')

380 case (compiler_f18)

382 case (compiler_ibmxl)
383 i = index(string, 'version:')
384 info = compilerinfo(id, 'xlf90', 'IBM', trim(adjustl(string(i+8:))))

386 info = compilerinfo(id, 'crayftn','CRAY', 'X.X.X.X')
