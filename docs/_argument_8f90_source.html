125 subroutine any_assign_argument(lhs, rhs)
126 class(arg), intent(inout) :: lhs
127 class(*), intent(in) :: rhs
128

130 select type (rhs)
131 type is (arg)

134 allocate(lhs%value, source=rhs%value)
135 class default
136 if (allocated(lhs%display)) deallocate(lhs%display)
137 allocate(lhs%value, source=rhs)
